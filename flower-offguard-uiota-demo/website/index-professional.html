Python - Start Federated Learning

import lltoken_sdk

# Initialize the client
client = lltoken_sdk.FederatedLearningClient(
    api_key="your_api_key",
    endpoint="https://api.lltoken.enterprise"
)

# Start training session
session = client.start_training(
    model_type="facial_recognition",
    min_clients=10,
    rounds=100,
    dataset_config={
        "privacy_level": "high",
        "encryption": "quantum_safe"
    }
)

print(f"Training session started: {session.id}")
